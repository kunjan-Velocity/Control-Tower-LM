function parseDate(s){ if(!s||s==='-'||String(s).trim()==='') return null; if(s instanceof Date){var d2=new Date(s);d2.setHours(0,0,0,0);return d2;} s=String(s).trim(); // Only trust new Date() for unambiguous ISO format: YYYY-MM-DD or YYYY/MM/DD // (4-digit year first). All other formats go straight to manual parsing // so that DD-MM-YY (Indian standard like 01-06-26 = 1 June) is never // mis-read as MM-DD-YY by the browser. if(/^\d{4}[-\/]\d{1,2}[-\/]\d{1,2}/.test(s)){ var d=new Date(s); if(!isNaN(d)){d.setHours(0,0,0,0);return d;} } var pts=s.split(/[\/\-\.]/); if(pts.length===3){ var a=pts[0].trim(),b=pts[1].trim(),c2=pts[2].trim(),d; if(a.length===4){ // YYYY-MM-DD d=new Date(a+'-'+b.padStart(2,'0')+'-'+c2.padStart(2,'0')); } else if(c2.length===4){ // DD-MM-YYYY d=new Date(c2+'-'+b.padStart(2,'0')+'-'+a.padStart(2,'0')); } else { // DD-MM-YY → treat as 20YY (Indian standard: day first, then month) d=new Date('20'+c2+'-'+b.padStart(2,'0')+'-'+a.padStart(2,'0')); } if(!isNaN(d)){d.setHours(0,0,0,0);return d;} } return null; }
